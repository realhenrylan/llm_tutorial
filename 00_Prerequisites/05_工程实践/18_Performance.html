📝 代码示例：数据加载优化

# 数据加载优化
import torch
from torch.utils.data import DataLoader, TensorDataset
import time

# 创建大数据集
x = torch.randn(10000, 100)
y = torch.randint(0, 10, (10000,))
dataset = TensorDataset(x, y)

# 方法1: 基本DataLoader（慢）
print("📦 方法1: 基本DataLoader")
dataloader1 = DataLoader(dataset, batch_size=32, num_workers=0)
start = time.time()
for batch in dataloader1:
    pass
print(f"  耗时: {time.time() - start:.4f}s")

# 方法2: 多进程加载（快）
print("\n📦 方法2: 多进程加载")
dataloader2 = DataLoader(dataset, batch_size=32, num_workers=4)
start = time.time()
for batch in dataloader2:
    pass
print(f"  耗时: {time.time() - start:.4f}s")

# 方法3: 多进程+预取（更快）
print("\n📦 方法3: 多进程+预取")
dataloader3 = DataLoader(dataset, batch_size=32, num_workers=4, pin_memory=True)
start = time.time()
for batch in dataloader3:
    pass
print(f"  耗时: {time.time() - start:.4f}s")

# 方法4: 自定义Dataset
print("\n📦 方法4: 自定义Dataset")
class FastDataset(TensorDataset):
    def __init__(self, *tensors):
        super().__init__(*tensors)
        self.data = tensors[0]
        self.labels = tensors[1]
    
    def __getitem__(self, idx):
        return self.data[idx], self.labels[idx]

fast_dataset = FastDataset(x, y)
dataloader4 = DataLoader(fast_dataset, batch_size=32, num_workers=4, pin_memory=True)
print(""  ✅ 自定义Dataset已创建")

🔥 JIT编译: 即时编译优化代码

📝 代码示例：计算优化

# 计算优化
import torch
import time

# 1. CPU vs GPU对比
print("🔢 1. CPU vs GPU对比")
x = torch.randn(1000, 1000)
y = torch.randn(1000, 1000)

if torch.cuda.is_available():
    x_gpu = x.to('cuda')
    y_gpu = y.to('cuda')
    
    print("  GPU可用")
    
    # CPU计算
    start = time.time()
    z_cpu = torch.matmul(x, y)
    print(f"  CPU耗时: {time.time() - start:.4f}s")
    
    # GPU计算
    start = time.time()
    z_gpu = torch.matmul(x_gpu, y_gpu)
    torch.cuda.synchronize()
    print(f"  GPU耗时: {time.time() - start:.4f}s")
    print(f"  加速比: {(time.time() - start)/0.01:.1f}x")
else:
    print("  GPU不可用，跳过GPU测试")

# 2. JIT编译优化
print("\n🔢 2. JIT编译优化")
def my_func(x):
    return torch.relu(x) + torch.sin(x)

jit_func = torch.jit.script(my_func)

x = torch.randn(100000)

# 普通版本
start = time.time()
for _ in range(1000):
    my_func(x)
print(f"  普通版本耗时: {time.time() - start:.4f}s")

# JIT版本
start = time.time()
for _ in range(1000):
    jit_func(x)
print(f"  JIT版本耗时: {time.time() - start:.4f}s")

# 3. 算子融合
print("\n🔢 3. 算子融合")
print("  PyTorch会自动融合一些算子")
print("  使用torch.compile可以进一步优化")

# 4. 量化
print("\n🔢 4. 量化")
model = torch.nn.Linear(100, 10)
quantized_model = torch.quantization.quantize_dynamic(
    model, {torch.nn.Linear}, dtype=torch.qint8
)
print("  ✅ 模型已量化为int8")

💾 内存管理

通俗理解：内存管理就像整理房间，把不需要的东西扔掉，把需要的东西放好，让空间更宽敞。

优化方法：释放不需要的张量、使用内存映射、梯度检查点、避免内存泄漏

🎨 可视化演示：内存管理

💾 内存管理技巧
🗑️ del释放: 删除不需要的张量
🔄 detach(): 断开梯度追踪释放内存
📊 梯度检查点: 节省中间激活内存
💾 内存映射: 大文件按需加载

显存是宝贵资源，良好的内存管理可以训练更大的模型！

📝 代码示例：内存管理

# 内存管理
import torch

# 1. 查看GPU内存
print("💾 1. 查看GPU内存")
if torch.cuda.is_available():
    print("  总显存:", torch.cuda.get_device_properties(0).total_memory / 1e9, "GB")
    print("  已用显存:", torch.cuda.memory_allocated() / 1e9, "GB")
    print("  峰值显存:", torch.cuda.max_memory_allocated() / 1e9, "GB")
else:
    print("  GPU不可用")

# 2. 释放内存
print("\n💾 2. 释放内存")
x = torch.randn(1000, 1000, requires_grad=True)
y = x * 2
z = y.sum()
z.backward()

print("  x.grad存在:", x.grad is not None)

# 方法1: del释放
del x
print("  del x后")

# 方法2: detach释放梯度
y = torch.randn(1000, 1000, requires_grad=True)
y_detached = y.detach()
print("  y_detached.requires_grad:", y_detached.requires_grad)

# 3. 梯度检查点
print("\n💾 3. 梯度检查点")
def forward_fn(x):
    x = torch.nn.functional.relu(x)
    x = torch.nn.functional.linear(x, torch.randn(100, 100))
    return x

x = torch.randn(32, 100, requires_grad=True)

# 使用梯度检查点
output = torch.utils.checkpoint.checkpoint(forward_fn, x)
output.sum().backward()
print("  ✅ 梯度检查点已应用")

# 4. 清空缓存
print("\n💾 4. 清空缓存")
if torch.cuda.is_available():
    torch.cuda.empty_cache()
    print("  ✅ GPU缓存已清空")

# 5. 内存使用最佳实践
print("\n💾 5. 内存使用最佳实践")
print("  1. 使用较小的batch_size")
print("  2. 删除不需要的张量")
print("  3. 使用梯度检查点")
print("  4. 定期清空缓存")

✏️ 性能优化实战练习

通俗理解：现在我们来综合运用各种性能优化技巧，打造一个高效的训练流程。

在LLM中的作用：性能优化是训练大型语言模型的关键，可以大幅缩短训练时间。

📝 完整示例：高性能训练流程

# 高性能训练流程
import torch
import torch.nn as nn
import torch.optim as optim
from torch.utils.data import DataLoader, TensorDataset

class HighPerformanceTrainer:
    def __init__(self, model, device='cuda'):
        self.model = model.to(device)
        self.device = device
        self.model = torch.compile(model)
    
    def train(self, dataloader, criterion, optimizer, epochs=10):
        self.model.train()
        
        for epoch in range(epochs):
            self.model.train()
            
            for inputs, targets in dataloader:
                inputs, targets = inputs.to(self.device), targets.to(self.device)
                
                optimizer.zero_grad()
                
                with torch.autocast(device_type=self.device, dtype=torch.float16):
                    outputs = self.model(inputs)
                    loss = criterion(outputs, targets)
                
                loss.backward()
                optimizer.step()
            
            print(f"Epoch {epoch+1}完成")

print("✅ 高性能训练器已创建")
print("优化策略: GPU + torch.compile + 混合精度")
